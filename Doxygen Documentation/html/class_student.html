bool Student::RemoveClass(Request r) {
vector<UCClass>::iterator i;
i=classes_.begin();
for(UCClass u:classes_){
if(u.getUc()==r.getUc() && u.getClass()==r.getClassIn()){
classes_.erase(i);
return true;
}
i++;
}
return false;
}
